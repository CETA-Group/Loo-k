import express from "express";
import axios from "axios";
import { GoogleGenerativeAI } from "@google/generative-ai";

const app = express();
app.use(express.json());

const GEMINI_API_KEY = "YOUR_GEMINI_KEY";
const GOOGLE_MAPS_KEY = "YOUR_GOOGLE_MAPS_KEY";

const genAI = new GoogleGenerativeAI(GEMINI_API_KEY);

app.post("/search-places", async (req, res) => {

  const { interests, location } = req.body;

  try {

    const model = genAI.getGenerativeModel({ model: "gemini-pro" });

    const prompt = `
    Convert these user interests into Google Maps place search keywords.

    Interests:
    ${interests.join(", ")}

    Return only a JSON array of keywords.
    Example: ["gym", "park", "bar"]
    `;

    const result = await model.generateContent(prompt);
    const text = result.response.text();

    const keywords = JSON.parse(text);

    let places = [];

    for (const keyword of keywords) {

      const response = await axios.get(
        `https://maps.googleapis.com/maps/api/place/textsearch/json`,
        {
          params: {
            query: keyword,
            location: location,
            radius: 3000,
            key: GOOGLE_MAPS_KEY
          }
        }
      );

      const results = response.data.results.map(p => ({
        name: p.name,
        address: p.formatted_address,
        category: keyword
      }));

      places = places.concat(results);
    }

    const counts = {};

    places.forEach(p => {
      counts[p.category] = (counts[p.category] || 0) + 1;
    });

    res.json({
      counts,
      places
    });

  } catch (err) {
    console.error(err);
    res.status(500).send("Error processing request");
  }

});

app.listen(3001, () => {
  console.log("Server running on port 3001");
});
